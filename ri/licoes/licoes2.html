SELECT nvl(trim(r.ind_fraude), 'Outro') TIPO_FRAUDE, COUNT(R.COD_NOTA ) qtd

            SELECT trim(v.ind_fraude) TF_VISITA,

            select cod_nota, num_seq, id_cdc, maior_peso, maior_resultado as resultado,
                case when menor_resultado in ('ATIRREGT01','ATIRREGT02')
                    then menor_ind_fraude
                else maior_ind_fraude end ind_fraude,
                maior_descricao,
                case   when (menor_resultado in ('ATIRREGT01','ATIRREGT02') or maior_ind_fraude in ('FR','IR'))
                    then rank() over (partition by cod_nota order by maior_peso desc, num_seq desc)
                    else 0
                end as num_rank
                ,maior_ind_fraude ind_fraude_T2
            from (select r2.cod_nota, id_cdc, r2.num_seq,
                t2.peso maior_peso,
                r2.resultado maior_resultado,
                t2.descricao maior_descricao,
                t2.ind_fraude maior_ind_fraude,
                first_value(r2.resultado)  over (partition by cod_nota, num_seq order by peso    asc  nulls last) menor_resultado,
                first_value(t2.ind_fraude) over (partition by cod_nota, num_seq order by peso    asc  nulls last) menor_ind_fraude,
                row_number()               over (partition by cod_nota, num_seq order by peso    desc nulls last) linha
                from prj_cicop.resultado_insp r2
                join prj_cicop.tipo_resultado_insp_peso t2 on r2.resultado = t2.resultado
                join (select id_cdc
                    from ri_energy_light.cdc
                    where id_grupo = 'B') c on to_number(instalacao) = c.id_cdc
                where t2.peso is not null
                    and r2.cod_nota in ('010038043630','010038857200','010014647978','010037287578'))
