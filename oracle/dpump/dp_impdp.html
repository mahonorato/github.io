# O utilitário de importação DPIMP

# Gerando um arquivo com o conteúdo de um arquivo dmp gerado anteriormente:

impdp usuario/senha@banco DIRECTORY=nome_do_diretorio DUMPFILE=arquivo.dmp SQLFILE=arquivo_saida.sql

Exemplo:

  • Exportando os objetos:

expdp adm_banco@banco directory='DADOS_DIR' schemas=adm_banco dumpfile=bkp_diario_20190601.dmp logfile=bkp_diario_20190601.log

  • Gerando o arquivo bkp_diario_20190601.sql com o conteúdo do arquivo DUMP:

impdp adm_banco@banco directory='DADOS_DIR' dumpfile=bkp_diario_20190601.dmp sqlfile=bkp_diario_20190601.sql

# Importando uma tabela

impdp usuario/senha@banco DIRECTORY=nome_do_diretorio TABLES=nome_esquema.nome_tabela DUMPFILE=arquivo.dmp

Exemplo:

impdp adm_banco@banco DIRECTORY='DADOS_DIR' TABLES=adm_banco.pais DUMPFILE=bkp_diario_20190601.dmp

# Importando uma tabela para outro esquema

impdp usuario/senha@banco DIRECTORY=nome_do_diretorio CONTENT=data_only REMAP_SCHEMA=esquema_origem:esquema_destino LOGFILE=nome-arquivo.log DUMPFILE=nome-arquivo.dmp

Exemplo:

impdp adm_banco/senha@banco directory='DATAPUMP_DIR' content=data_only remap_schema=meu_esquema:adm_banco logfile=importa_para_adm_banco.log dumpfile=tabelas_meu_esquema_201910.dmp

# Importando tabelas de um esquema para outro usando DBLINK

impdp usuario/senha@banco DIRECTORY=nome_do_diretorio SCHEMAS=esquema-origem REMAP_SCHEMA=esquema-origem:esquema-destino CONTENT=METADATA_ONLY NETWORK_LINK=meu_dblink LOGFILE=importacao-esquema.LOG
